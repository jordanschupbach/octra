message("Constructing Java bindings...")

cmake_minimum_required(VERSION 3.18)
if(POLICY CMP0122)
  cmake_policy(SET CMP0122 NEW)
endif()

project(joctra VERSION 0.1)

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON CACHE INTERNAL "")
if(CMAKE_EXPORT_COMPILE_COMMANDS)
  set(CMAKE_CXX_STANDARD_INCLUDE_DIRECTORIES
      ${CMAKE_CXX_IMPLICIT_INCLUDE_DIRECTORIES})
endif()

file(COPY ${PROJECT_SOURCE_DIR}/../../include/octra DESTINATION ${PROJECT_BINARY_DIR}/)
file(COPY ${PROJECT_SOURCE_DIR}/../../src/octra DESTINATION ${PROJECT_BINARY_DIR}/)
file(COPY ${PROJECT_SOURCE_DIR}/../../include/ DESTINATION ${PROJECT_BINARY_DIR}/include)
file(COPY ${PROJECT_SOURCE_DIR}/../../src/octra/ DESTINATION ${PROJECT_BINARY_DIR}/source)

file(GLOB_RECURSE C_HEADERS "${PROJECT_BINARY_DIR}/*.h")
file(GLOB_RECURSE CXX_HEADERS "${PROJECT_BINARY_DIR}/*.hpp")
file(GLOB_RECURSE C_SOURCES "${PROJECT_BINARY_DIR}/*.c")
file(GLOB_RECURSE CXX_SOURCES "${PROJECT_BINARY_DIR}/*.cpp")
set(OCTRA_SOURCES ${C_HEADERS} ${CXX_HEADERS} ${C_SOURCES} ${CXX_SOURCES})

file(COPY ${PROJECT_SOURCE_DIR}/swig_src DESTINATION ${PROJECT_BINARY_DIR}/)

message("Building SWIG interface...")
execute_process(COMMAND swig -c++ -java -o ${PROJECT_BINARY_DIR}/swig_src/octra_wrap.cpp ${PROJECT_BINARY_DIR}/swig_src/joctra.i 
  WORKING_DIRECTORY ${PROJECT_BINARY_DIR})
message("Done building SWIG interface...")

file(GLOB_RECURSE C_SOURCES "${PROJECT_SOURCE_DIR}/../../src/octra/*.c")
set(QUOTED_C_SOURCES "")
foreach(filepath ${C_SOURCES})
string(REPLACE "${PROJECT_SOURCE_DIR}/../../src/octra/" "src/joctra/octra/" relative_filepath ${filepath})
    set(QUOTED_C_SOURCES "${QUOTED_C_SOURCES}'${relative_filepath}', ")
endforeach()

file(GLOB_RECURSE CXX_SOURCES "${PROJECT_SOURCE_DIR}/../../src/octra/*.cpp")
set(QUOTED_CXX_SOURCES "")
foreach(filepath ${CXX_SOURCES})
message("${filepath}")
    string(REPLACE "${PROJECT_SOURCE_DIR}/../../src/octra/" "src/joctra/octra/" relative_filepath ${filepath})
    set(QUOTED_CXX_SOURCES "${QUOTED_CXX_SOURCES}'${relative_filepath}', ")
endforeach()

# message("QUOTED_C_SOURCES: ${QUOTED_C_SOURCES}")
# message("QUOTED_CXX_SOURCES: ${QUOTED_CXX_SOURCES}")
# message("Project Source Directory: ${PROJECT_SOURCE_DIR}")
# message("Project Bin Directory: ${PROJECT_BINARY_DIR}")

file(COPY ${PROJECT_BINARY_DIR}/include/ DESTINATION ${PROJECT_BINARY_DIR}/../../src/joctra-octra/octra/src/)
file(COPY ${PROJECT_BINARY_DIR}/source/ DESTINATION ${PROJECT_BINARY_DIR}/../../src/joctra-octra/octra/src/)
file(COPY ${PROJECT_BINARY_DIR}/swig_src/octra_wrap.cpp DESTINATION ${PROJECT_BINARY_DIR}/../../src/joctra-octra/octra/src/source/octra/)

file(GLOB JAVA_FILES "${PROJECT_BINARY_DIR}/swig_src/*.java")
foreach(JAVA_FILE ${JAVA_FILES})
    get_filename_component(FILENAME ${JAVA_FILE} NAME)
    configure_file(${JAVA_FILE} "${PROJECT_BINARY_DIR}/../../src/joctra/src/main/java/js/octra/joctra/${FILENAME}" COPYONLY)
endforeach()
